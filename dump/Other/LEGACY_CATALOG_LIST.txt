Select distinct
Item_Cat.Inventory_Item_Id, 
Item_Cat.Organization_Id, 
upper(Item_Cat.Category_Set_Name),
Itemwarehouse.Product_Number, 
Item_Cat.Segment1, 
Item_Cat.Segment2, 
Item_Cat.Segment3, 
Item_Cat.Segment4, 
Item_Cat.Segment5, 
Item_Cat.Segment6, 
Item_Cat.Segment7, 
Item_Cat.Segment8, 
Item_Cat.Segment9, 
Item_Cat.Segment10

From 
IQP_ITEM_CATEGORY_DIM Item_Cat,
IQP_ITEMWAREHOUSE_DIM Itemwarehouse
Where 
Item_Cat.Inventory_Item_Id = Itemwarehouse.Inventory_Item_Id
and Item_Cat.Organization_Id = Itemwarehouse.Organization_Id
and Item_Cat.Organization_Id in ('2453','2454','3282','3283','3284','3719')
and decode(Item_Cat.Organization_Id, '2453', 'Order Entry Category', '2454', 'Order Entry Category', '3282', 'Product Line Category',	'3283', 'Product Line Category',
'3284', 'Product Line Category', '3719', 'ROC Product Line Category', Item_Cat.Category_Set_Name) = Item_Cat.Category_Set_Name
and Item_Cat.Category_Set_Name in ('Product Line Category', 'ROC Product Line Category', 'Order Entry Category')

Order by 1, 2
